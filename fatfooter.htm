Fat-footer Mixin

                                
$fatBgColor: #FFF;                        //fatFooter 底色
$fatTextColor: #333;                      //fatFooter 連結顏色
.FatFooter {
//  @include make-fatfooter(欄位數,nav最大的寬度)
    @include make-fatfooter(6,1000px);    //設定fatfooter欄，手機版預設隱藏
}

Fat-footer Jquery設定

                                
/*-----------------------------------*/
/////////////fatfooter開關/////////////
/*-----------------------------------*/
$(document).ready(function() {
    $(".btn-fatfooter").click(function() {
        $('.fatfooter>nav>ul>li>ul').stop(true,true).slideToggle(function() {
            if ($(this).is(':visible')) {
                $(".btn-fatfooter").html("收合");
                $(".btn-fatfooter").attr('name', '收合選單');
            } else {
                $(".btn-fatfooter").html("展開");
                $(".btn-fatfooter").attr('name', '展開選單');
            }
        });
        $(this).stop(true,true).toggleClass('close');
    });
});
