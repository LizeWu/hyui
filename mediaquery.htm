@mixin screen($size) {
  $desktop:     "(min-width: #{$screen-lg})";               // max-width大於1400px
  $notebook:    "(max-width: #{$screen-lg-minWidth})";      // max-width小於1399px
  $tablet:      "(max-width: #{$screen-md})";               // max-width小於992px
  $mobile:      "(max-width: #{$screen-sm})";               // max-width小於768px
  $xs_mobile:   "(max-width: #{$screen-xs})";               // max-width小於576px

  @if $size == desktop {
    @media screen and #{$desktop} {
      @content;
    }
  }
  @else if $size == notebook {
    @media screen and #{$notebook} {
      @content;
    }
  }
  @else if $size == tablet {
    @media screen and #{$tablet} {
      @content;
    }
  }
  @else if $size == mobile {
    @media screen and #{$mobile} {
      @content;
    }
  }
   @else if $size == xs_mobile {
    @media screen and #{$xs-mobile} {
      @content;
    }
  }
  @else {
    @media screen and #{$size} {
      @content;
    }
  }
}
